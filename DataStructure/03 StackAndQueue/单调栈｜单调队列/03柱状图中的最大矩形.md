class Solution:
    def largestRectangleArea(self, heights: List[int]) -> int:
        # 对于每个柱子，向左、向右找到第一个比自己矮的
        # 目的是去看，包含每个柱子并且以该柱子高度为高度的矩形最宽有多宽
        # 做法就是前面一道hot100题: 739. 每日温度

        # p.s. 最终答案的矩形一定是和其中某一(几)个柱子的高度一样，因此这么做是不会漏掉答案的
        l = len(heights)
        right = [l-i-1 for i in range(l)]
        stack = []
        for i in range(l):
            h = heights[i]
            while stack and heights[stack[-1]] > h:
                idx = stack.pop(-1)
                right[idx] = i-1-idx
            stack.append(i)
        
        left = [i for i in range(l)]
        stack = []
        for i in reversed(range(l)):
            h = heights[i]
            while stack and heights[stack[-1]] > h:
                idx = stack.pop(-1)
                left[idx] = idx-1-i
            stack.append(i)
        
        ans = heights[0]
        for i in range(l):
            # print(left[i], right[i], heights[i])
            ans = max(ans, (left[i]+right[i]+1)*heights[i])
        return ans
