class Solution:
    def dailyTemperatures(self, temperatures: List[int]) -> List[int]:
        stack = []
        l = len(temperatures)
        ans = [0 for _ in range(l)]
        for i, t in enumerate(temperatures):
            while stack and temperatures[stack[-1]] < t:
                idx = stack.pop(-1)
                ans[idx] = i-idx
            stack.append(i)
        return ans
