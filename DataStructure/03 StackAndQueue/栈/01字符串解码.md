class Solution:
    def decodeString(self, s: str) -> str:
        stack = []
        for e in s:
            if e == ']':
                num = k = 0
                word = ''
                while stack[-1] != '[':
                    p = stack.pop(-1)
                    word = p + word
                stack.pop(-1)
                while stack and stack[-1] in '0123456789':
                    p = stack.pop(-1)
                    num += int(p) * (10 ** k)
                    k += 1
                stack.append(word*num)
            else:
                stack.append(e)
        ans = ''
        for e in stack: ans += e
        return ans
