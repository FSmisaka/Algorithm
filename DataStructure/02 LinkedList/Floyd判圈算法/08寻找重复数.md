```python
class Solution:
    def findDuplicate(self, nums: List[int]) -> int:
        # 这题完全和 142. 环形链表II 一样
        # 重复的数 == 环形链表的入口前一个节点
        # 之所以有环, 就是因为这多个重复的数都指向了环入口
        slow = fast = nums[0]
        while nums[nums[fast]] != nums[slow]:
            fast = nums[nums[fast]]
            slow = nums[slow]
        ans = 0
        while ans != slow:
            ans = nums[ans]
            slow = nums[slow]
        return ans
```
