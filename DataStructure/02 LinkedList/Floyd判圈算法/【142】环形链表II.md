```python
# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, x):
#         self.val = x
#         self.next = None

class Solution:
    def detectCycle(self, head: Optional[ListNode]) -> Optional[ListNode]:
        # 看 0神 的题解，里面有个图更清晰一点
        # 假设从头节点到环入口为a，快慢指针相遇时分别走了2b和b步，环长为c，环入口到相遇点走了d
        # 首先一定有2b-b=k_1*c，因为快指针比慢指针肯定是多走了环长的倍数，这个k_1是假设的倍数
        # 化简等式b=k_1*c，所以实际上慢指针相当于走了k_1圈走到了相遇点
        # 而慢指针走的步数b=a+d+k_2*c，因为慢指针肯定是走到相遇点并绕了几圈（可能是0圈），假设这个圈数是k_2
        # 那么a+d+k_2*c = k_1*c，于是发现a+d，即从头节点走到相遇点，也是环长c的倍数
        # a = (k_1-k_2)*c-d = c-d+(k_1-k_2-1)*c
        # 1. k_1 == k_2，先暂时不讨论
        # 2. k_1 > k_2
        #    头节点走到环入口的步数a = 从相遇点走k_1-k_2-1圈之后再走c-d
        #    因此只需要让头节点和相遇点同时开走就能在环入口相遇
        #    而这个策略刚好适合于前面未讨论的情况1，因为头节点和相遇点在最开始就已经相遇在环入口了
        
        slow = fast = head
        while slow and fast and fast.next:
            fast = fast.next.next
            slow = slow.next
            if slow is fast:  # 有环
                while not head is slow:
                    head = head.next
                    slow = slow.next
                return head
        return None
```
