```python
# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, x):
#         self.val = x
#         self.next = None

class Solution:
    def getIntersectionNode(self, headA: ListNode, headB: ListNode) -> Optional[ListNode]:
        def get_len(node):
            l = 0
            cur = node
            while cur:
                l += 1
                cur = cur.next
            return l
        lA, lB = get_len(headA), get_len(headB)
        if lA >= lB:
            for _ in range(lA-lB): headA = headA.next
        else:
            for _ in range(lB-lA): headB = headB.next
        
        while headA and headB:
            if headA is headB: return headA
            headA = headA.next
            headB = headB.next
        return None
```
