# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next
class Solution:
    def isPalindrome(self, head: Optional[ListNode]) -> bool:
        l = 0
        cur = head
        while cur:
            l += 1
            cur = cur.next
        
        cur = head
        for _ in range(math.ceil(l/2)): cur = cur.next

        def reverse_list(node):
            pre = None
            while node:
                tmp = node.next
                node.next = pre
                pre = node
                node = tmp
            return pre

        cur1 = head
        cur2 = reverse_list(cur)
        while cur1 and cur2:
            if not cur1.val == cur2.val: return False
            cur1 = cur1.next
            cur2 = cur2.next
        return True
