```python
class Solution:
    def longestConsecutive(self, nums: List[int]) -> int:
        num_set = set(nums)
        ans = 0
        for num in num_set:
            if num-1 in num_set: continue
            k = 1
            while num+k in num_set:
                k += 1
            ans = max(ans, k)
        return ans
```
