class Solution:
    def longestValidParentheses(self, s: str) -> int:
        dp = []
        ans = 0
        for i, p in enumerate(s):
            # 'xxxx()'
            if p == ')' and i-1 >= 0 and s[i-1] == '(':
                dp.append(2+(dp[i-2] if i-2 >= 0 else 0))
            # '((xxxxxx))'
            elif p == ')' and i-1 >= 0 and s[i-1] == ')' and i-1-dp[i-1] >= 0 and s[i-1-dp[i-1]] == '(':
                dp.append(dp[i-1]+2+(dp[i-1-dp[i-1]-1] if i-1-dp[i-1]-1 >= 0 else 0))
            else: dp.append(0)
            ans = max(ans, dp[-1])
        # print(dp)
        return ans
