class Solution:
    def lengthOfLIS(self, nums: List[int]) -> int:
        dp = []
        tails = {}
        max_pre = 1
        for i, num in enumerate(nums):
            if i == 0:
                dp.append(1)
                tails[1] = num
                continue
            max_len = 1
            left, right = 1, max_pre
            while left <= right:
                mid = (left + right) // 2
                if tails[mid] < num:
                    max_len = max(max_len, mid+1)
                    left = mid+1
                else:
                    right = mid-1
            max_pre = max(max_pre, max_len)
            tails[max_len] = min(tails[max_len], num) if tails.get(max_len, None) else num
            dp.append(max_len)
        return max_pre
