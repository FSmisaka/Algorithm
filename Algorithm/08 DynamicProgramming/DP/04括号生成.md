```python
class Solution:
    def generateParenthesis(self, n: int) -> List[str]:
        if n == 1: return ['()']
        ret = []
        for ps in self.generateParenthesis(n-1):
            cnt = 0
            for i, s in enumerate(ps):
                if cnt == 0: ret.append('('+ps[:i]+')'+ps[i:])
                if s == '(': cnt += 1
                else: cnt -= 1
            ret.append('('+ps+')')
        return ret
```
