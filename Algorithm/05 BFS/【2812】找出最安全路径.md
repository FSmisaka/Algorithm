## 题目

路径最小值最大化问题

## DFS解法 (TLE)

```python
class Solution:
    def maximumSafenessFactor(self, grid: List[List[int]]) -> int:
        m, n = len(grid), len(grid[0])
        # 多源
        visited = set()
        q = []
        for i in range(m):
            for j in range(n):
                if grid[i][j] == 1:
                    q.append((i, j))
                    visited.add((i, j))
                    grid[i][j] = 0
        
        # BFS
        danger = 1
        while q:
            nxt = []
            while q:
                i, j = q.pop()
                nxt_step = [(1, 0), (-1, 0), (0, 1), (0, -1)]
                for stp_i, stp_j in nxt_step:
                    if (0 <= i+stp_i < m and 0 <= j+stp_j < n and (i+stp_i, j+stp_j) not in visited):
                        nxt.append((i+stp_i, j+stp_j))
                        visited.add((i+stp_i, j+stp_j))
                        grid[i+stp_i][j+stp_j] = danger
            q += nxt
            danger += 1
        
        # DFS + 二分
        def dfs(i, j, k, visited):
            nonlocal m, n
            if i == m-1 and j == n-1: return True
            if not (0 <= i < m and 0 <= j < n and grid[i][j] >= k): return False
            visited.add((i, j))
            nxt_step = [(1, 0), (-1, 0), (0, 1), (0, -1)]
            flag = False
            for stp_i, stp_j in nxt_step:
                if (i+stp_i, j+stp_j) in visited: continue
                flag = flag or dfs(i+stp_i, j+stp_j, k, visited)
            visited.remove((i, j))
            return flag

        left, right = 0, min(grid[0][0], grid[-1][-1])
        while left <= right:
            mid = (left + right) // 2
            if dfs(0, 0, mid, set()): left = mid+1
            else: right = mid-1
        
        return right
```

## BFS解法

```python
class Solution:
    def maximumSafenessFactor(self, grid: List[List[int]]) -> int:
        m, n = len(grid), len(grid[0])
        # 多源
        visited = set()
        q = []
        for i in range(m):
            for j in range(n):
                if grid[i][j] == 1:
                    q.append((i, j))
                    visited.add((i, j))
                    grid[i][j] = 0
        
        # BFS
        danger = 1
        while q:
            nxt = []
            while q:
                i, j = q.pop()
                nxt_step = [(1, 0), (-1, 0), (0, 1), (0, -1)]
                for stp_i, stp_j in nxt_step:
                    if (0 <= i+stp_i < m and 0 <= j+stp_j < n and (i+stp_i, j+stp_j) not in visited):
                        nxt.append((i+stp_i, j+stp_j))
                        visited.add((i+stp_i, j+stp_j))
                        grid[i+stp_i][j+stp_j] = danger
            q += nxt
            danger += 1
        
        # BFS + 二分
        def bfs(k):
            if grid[0][0] < k: return False
            q = [(0, 0)]
            visited = set([(0, 0)])

            while q:
                nxt = []
                while q:
                    i, j = q.pop()
                    nonlocal m, n
                    if i == m-1 and j == n-1: return True
                    nxt_step = [(1, 0), (-1, 0), (0, 1), (0, -1)]
                    for stp_i, stp_j in nxt_step:
                        nxt_id = (i+stp_i, j+stp_j)
                        if (0 <= nxt_id[0] < m and 0 <= nxt_id[1] < n and nxt_id not in visited and grid[nxt_id[0]][nxt_id[1]] >= k):
                            visited.add(nxt_id)
                            nxt.append(nxt_id)
                q += nxt
            return False

        left, right = 0, min(grid[0][0], grid[-1][-1])
        while left <= right:
            mid = (left + right) // 2
            if bfs(mid): left = mid+1
            else: right = mid-1
        
        return right
```
