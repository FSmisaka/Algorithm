class Solution:
    def firstMissingPositive(self, nums: List[int]) -> int:
        l = len(nums)
        i = 0
        while i < l:
            if nums[i] <= 0 or nums[i] > l or nums[i] == i+1 or nums[nums[i] - 1] == nums[i]:
                i += 1
                continue
            tmp = nums[nums[i] - 1]
            nums[nums[i] - 1] = nums[i]
            nums[i] = tmp
        
        i = 0
        while i < l:
            if not i+1 == nums[i]: 
                return i+1
            i += 1
        return i+1
