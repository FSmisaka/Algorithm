```python
class Solution:
    def productExceptSelf(self, nums: List[int]) -> List[int]:
        ans = [1]
        for num in nums:
            ans.append(ans[-1]*num)
        l = len(nums)
        st = 1
        for i in range(l):
            j = l-i-1
            ans[j] *= st
            st *= nums[j]
        return ans[:-1]
```
