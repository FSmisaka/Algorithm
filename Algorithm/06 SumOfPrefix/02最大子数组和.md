```python
class Solution:
    def maxSubArray(self, nums: List[int]) -> int:
        pre = [0]
        for num in nums: pre.append(pre[-1]+num)
        min_num = 0
        ans = -float('inf')
        for num in pre[1:]:
            ans = max(ans, num-min_num)
            min_num = min(min_num, num)
        return ans
```
